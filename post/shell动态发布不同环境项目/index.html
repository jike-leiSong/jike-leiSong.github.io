shell动态发布不同环境项目

2018-01-25

build.sh

#!/usr/bin/env bash
#################################
# 源码路径
DIR=`pwd`
# 工程名
FINALNAME=ROOT
PORT=8089
SERVICE=fnw_manage_api
#################################

WAR=$DIR/target/$FINALNAME.war
# 构建生产环境
function build_dev(){
    git pull
    server_dir=/home/service/$SERVICE/
    mvn clean package -Dmaven.test.skip=true -Pdev
    ps -ef |grep $SERVICE|awk '{print $2}'|xargs kill -9
    rm -rf $server_dir/$FINALNAME*
    cp $WAR $server_dir/webapps/
    cd $server_dir 
    bin/startup.sh
}

# 构建测试环境
function build_test(){
    git pull
    server_dir=/home/service/$SERVICE/
    mvn clean package -Dmaven.test.skip=true -Ptest
    ps -ef |grep $SERVICE|awk '{print $2}'|xargs kill -9
    rm -rf $server_dir/webapps/$FINALNAME*
    cp $WAR $server_dir/webapps/
    cd $server_dir 
    bin/startup.sh
}

# 构建到开发环境
function build_pro(){
    git pull
    server_dir=/home/service/$SERVICE/
    mvn clean package -Dmaven.test.skip=true -Ppro
    ps -ef |grep $SERVICE|awk '{print $2}'|xargs kill -9
    rm -rf $server_dir/webapps/$FINALNAME*
    cp $WAR $server_dir/webapps/
    cd $server_dir 
    bin/startup.sh
}

# 构建到集成此时环境
function build_stage(){
    git pull
    server_dir=/home/service/$SERVICE/
    mvn clean package -Dmaven.test.skip=true -Pstage
    ps -ef |grep $SERVICE|awk '{print $2}'|xargs kill -9
    rm -rf $server_dir/webapps/$FINALNAME*
    cp $WAR $server_dir/webapps/
    cd $server_dir
    bin/startup.sh
}

if [[ -n $1 ]] && [[ $1 = 'test' ]]; then
    # 构建测试环境包
    read -p "发布到 [测试] 环境 ? y/n :"  val
    if [[ -n $val ]] && [[ $val = 'y' ]]; then 
        build_test
    else 
        echo '取消操作'
    fi
elif [[ -n $1 ]] && [[ $1 = 'pro' ]]; then
    # 构建生产环境包
    read -p "发布到 [生产] 环境 ? y/n :"  val
    if [[ -n $val ]] && [[ $val = 'y' ]]; then 
        build_pro
    else 
        echo '取消操作'
    fi
elif [[ -n $1 ]] && [[ $1 = 'dev' ]]; then
    # 构建开发环境包
    read -p "发布到 [开发] 环境 ? y/n :"  val
    if [[ -n $val ]] && [[ $val = 'y' ]]; then 
        build_dev
    else 
        echo '取消操作'
    fi
elif [[ -n $1 ]] && [[ $1 = 'stage' ]]; then
    # 构建集成测试环境包
    read -p "发布到 [集成测试] 环境 ? y/n :"  val
    if [[ -n $val ]] && [[ $val = 'y' ]]; then
        build_stage
    else
        echo '取消操作'
    fi
    
else
        echo './build.sh [option]'
        echo '------------ option -------------- '
        echo 'dev  :     构建测试环境.'
        echo 'test :     构建测试环境.'
        echo 'stage:     构建集成测试环境.'
        echo 'pro  :     构建生产环境.'
fi
